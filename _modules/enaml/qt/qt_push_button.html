[docs]    def menu(self):
        """ Find and return the menu child for this widget.

        Returns
        -------
        result : QMenu or None
            The menu defined for this widget, or None if not defined.

        """
        widget = None
        for child in self.children():
            if isinstance(child, QtMenu):
                widget = child.widget()
        return widget
